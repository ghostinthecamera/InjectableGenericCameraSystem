// Script to obtain object info about an object that has a vtable at the start. Also prints inheritance hierarchy
function DisplayFullClassName(address, currentName)
  if input==0 then
     return
  end
  local currentAddress = address
  local className = readString( readQword( currentAddress + 0x18))
  currentAddress = readQword(address+0x40)
  if string.len(currentName) > 0 then
    className = currentName .. "->" .. className
  end
  if currentAddress==0 then
     -- display full name
     print( string.format( "Class name:\t\t%s", className ) )
     return
  end
  DisplayFullClassName(currentAddress, className)
end

function GetName( address )
  local currentAddress = readQword(address)
  currentAddress = currentAddress + 0x38
  currentAddress = readQword( currentAddress )
  if readBytes( currentAddress, 1 ) == 0xE9 then
    currentAddress = currentAddress + readInteger( currentAddress + 0x1, true ) + 0x5
  end
  currentAddress = currentAddress + readInteger( currentAddress + 0x3, true ) + 0x7
  currentAddress = readQword( currentAddress )
  print( string.format("Struct address:\t\t0x%X", address))
  DisplayFullClassName(currentAddress, "")
  print( string.format("RTTIDataCompound:\t\t0x%X", currentAddress))
  print( string.format( "Object Hash:\t\t0x%X", readInteger(currentAddress + 0x24)))
  print("-----------------------------------")
end

  if currentAddress==nil or currentAddress==0 then

  return DisplayFullClassName(currentAddress, className)

function GetName( address )
  local currentAddress = address
  if currentAddress==nil then
     return ""
  end
  currentAddress = currentAddress + 0x38
  currentAddress = readQword( currentAddress )
  if currentAddress==nil then
    return ""
  end
  if readBytes( currentAddress, 1 ) == 0xE9 then
    currentAddress = currentAddress + readInteger( currentAddress + 0x1, true ) + 0x5
  end
  local intOffset = readInteger( currentAddress + 0x3, true )
  if intOffset==nil then
    return ""
  end
  currentAddress = currentAddress + intOffset + 0x7
  if currentAddress==nil then
     return ""
  end
  currentAddress = readQword( currentAddress )
  if currentAddress==nil then
     return ""
  end
  return string.format("Struct address: 0x%X | Type name: %s", address, DisplayFullClassName(currentAddress, ""))
end

function GetNames(startAddress, length)
  local offset=0
  while offset < length do
    local currentValue = readQword(startAddress + offset)
    local className = "Unknown"
    if currentValue > 0x7FF60000000 then
       className = GetName(currentValue)
       if string.len(className) > 0 then
         print(string.format("Offset: %06x: Reference type:\t%s", offset, className))
       end
    else
       if currentValue > (startAddress - 0x100000) and currentValue < (startAddress+0x100000) then
          local pointerAddress = currentValue
          currentValue = readQword(currentValue)
          className = GetName(currentValue)
          if string.len(className) > 0 then
            print(string.format("Offset: %06x: Pointer to 0x%X of type\t%s", offset, pointerAddress, className))
          end
       else
          -- nothing
          --print(string.format("Offset: %d: nothing of interest", offset))
       end
    end
    offset=offset+8
  end
end
